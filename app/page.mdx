```sql
CREATE TABLE animals (
  id integer PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
  slug varchar(80) UNIQUE NOT NULL,
  name varchar(80) NOT NULL
);

SELECT
  *
FROM
  animals;
```
